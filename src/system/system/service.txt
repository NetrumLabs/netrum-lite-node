[Unit]
Description=Netrum Node System Service - Auto Refresh Every 1 Minute
After=network-online.target
Wants=network-online.target

[Service]
Type=simple
User=root
WorkingDirectory=/root/netrum-lite-node/src/system/system
ExecStart=/usr/bin/node /root/netrum-lite-node/src/system/system/system.js
Restart=always
RestartSec=5s
StartLimitIntervalSec=60
StartLimitBurst=10
StandardOutput=journal
StandardError=journal

# Environment Variables
Environment=NODE_ENV=production
Environment=PYTHONUNBUFFERED=1
Environment=SPEED_TEST_INTERVAL=60000

# Limit service run to 65 seconds max (refresh every minute)
RuntimeMaxSec=65

[Install]
WantedBy=multi-user.target
