[Unit]
Description=Netrum Node Mining Synchronization Service
After=network.target

[Service]
Type=simple
User=root
WorkingDirectory=/root/netrum-lite-node
ExecStart=/usr/bin/node /root/netrum-lite-node/src/system/mining/start-mining.js
Restart=always
Environment=NODE_ENV=production
Environment=PYTHONUNBUFFERED=1  # Affects some Node.js versions
StandardOutput=journal
StandardError=journal
# Disable all rate limiting
LogRateLimitIntervalSec=0
LogRateLimitBurst=0
# Force immediate flushing
LogLevelMax=debug

[Install]
WantedBy=multi-user.target
